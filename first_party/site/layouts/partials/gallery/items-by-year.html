The returned list is sorted by year in descending order. */}} {{ $itemsByYearMap := dict }} {{ range site.Data.gallery.items }} {{ $yearKey := printf "%v" (int .date.year) }} {{ $dateStr := printf "%04d-%02d-%02d" (int .date.year) (int (.date.month | default 1)) (int (.date.day | default 1)) }} {{ $itemWithSort := merge .
